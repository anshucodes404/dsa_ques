class Solution {
    public int longestOnes(int[] nums, int k) {
        int maxLen = 0;
        int l = 0;
        int r = 0;
        int z = 0; // this will count the zeros in the current window

        while(r < nums.length){
            if(nums[r] == 0) z++;
            if(z >= k) {
                // reduce the window size to fit the zeros to only k
                while(z != k){
                    if(nums[l] == 0) z--;
                    l++;
                }
            }
            
            int len = r - l + 1;
            maxLen = Math.max(len, maxLen);
            r++;
        }
        return maxLen;
    }
}
